Public Sub keke()
DoEvents
Application.ScreenUpdating = True
Dim s As String
For i = 2 To Sheets("Outstanding Sales Order Status").UsedRange.Rows.Count
s = Sheets("Outstanding Sales Order Status").Cells(i, 5)
Dim cs As String
For k = 2 To Sheets("Sheet1").UsedRange.Rows.Count
cs = Sheets("Sheet1").Cells(k, 1)
If s = cs Then
Sheets("Outstanding Sales Order Status").Cells(i, 12) = Sheets("Sheet1").Cells(k, 2)
Sheets("Outstanding Sales Order Status").Cells(i, 13) = Sheets("Sheet1").Cells(k, 3)
k = Sheets("Sheet1").UsedRange.Rows.Count
End If
Next k
Next i
End Sub
