Sub Quarter()
DoEvents
Application.ScreenUpdating = True
 Dim WB1 As Workbook



Set WB1 = ActiveWorkbook

 
Dim ws As Worksheet

For Each ws In WB1.Worksheets
Dim name As String
name = ws.name
    WB1.Activate
    Sheets(name).Select
    Sheets(name).Columns("E").Select
    
    Dim item As String
    Dim length As Integer
    
    length = Sheets(name).UsedRange.Rows.count
    
    For i = 2 To length
    
    item = WB1.Sheets(name).Cells(i, "E")
      Dim btn As Button
      Dim t As Range
    Set t = ActiveSheet.Range(Cells(i, "W"), Cells(i, "W"))
    Set btn = ActiveSheet.Buttons.Add(t.Left, t.Top, t.Width, t.Height)
    With btn
      .OnAction = "btnS"
      .Caption = ""
      .name = item
    End With
 
    
    'need to switch to other wb, seacrh/filter pivot table and then copy over to existing sheet
    
    
    Next i
    
    Next ws
End Sub
