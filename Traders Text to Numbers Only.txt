Sub Letters_Out()

'this method converts traders items into numbers
    Dim i As Integer
    Dim Original As String
    Dim NumOnly As String
    For k = 1 To Sheets("Sheet1").UsedRange.Rows.Count
    Original = Sheets("Sheet1").Cells(k, 1)
    NumOnly = ""
    For i = 1 To Len(Original)
        If IsNumeric(Mid(Original, i, 1)) Then
            NumOnly = NumOnly & Mid(Original, i, 1)
        End If
        If Mid(Original, i, 1) = "," Then
            NumOnly = NumOnly & " "
        End If
    Next i
    Sheets("Sheet1").Cells(k, 1) = NumOnly
    Next k
    
End Sub
